JavaScript Array Methods

1. push()

Adds one or more elements to the end of an array:

let arr = [1, 2, 3];
arr.push(4, 5);
console.log(arr); // [1, 2, 3, 4, 5]

2. pop()

Removes the last element from an array:

let arr = [1, 2, 3];
arr.pop();
console.log(arr); // [1, 2]

3. shift()

Removes the first element from an array:

let arr = [1, 2, 3];
arr.shift();
console.log(arr); // [2, 3]

4. unshift()

Adds one or more elements to the beginning of an array:

let arr = [1, 2, 3];
arr.unshift(0);
console.log(arr); // [0, 1, 2, 3]

5. concat()

Combines multiple arrays into one:

let arr1 = [1, 2];
let arr2 = [3, 4];
let arr3 = arr1.concat(arr2);
console.log(arr3); // [1, 2, 3, 4]

6. slice()

Returns a shallow copy of a portion of an array:

let arr = [1, 2, 3, 4, 5];
let newArr = arr.slice(1, 4);
console.log(newArr); // [2, 3, 4]

7. splice()

Changes the content of an array by removing or replacing existing elements:

let arr = [1, 2, 3, 4, 5];
arr.splice(2, 1, 6, 7);
console.log(arr); // [1, 2, 6, 7, 4, 5]

8. forEach()

Executes a provided function once for each array element:

let arr = [1, 2, 3];
arr.forEach((element) => console.log(element));

9. map()

Creates a new array populated with the results of calling a provided function on every element in the array:

let arr = [1, 2, 3];
let doubled = arr.map(x => x * 2);
console.log(doubled); // [2, 4, 6]

10. filter()

Creates a new array with all elements that pass the test implemented by the provided function:

let arr = [1, 2, 3, 4, 5];
let even = arr.filter(x => x % 2 === 0);
console.log(even); // [2, 4]

11. reduce()

Applies a function to reduce the array to a single value:

let arr = [1, 2, 3, 4];
let sum = arr.reduce((acc, val) => acc + val, 0);
console.log(sum); // 10

12. find()

Returns the first element in the array that satisfies the provided testing function:

let arr = [1, 2, 3, 4, 5];
let found = arr.find(x => x > 3);
console.log(found); // 4

13. indexOf()

Returns the first index at which a given element is found in the array, or -1 if it is not found:

let arr = [1, 2, 3, 4];
let index = arr.indexOf(3);
console.log(index); // 2

14. includes()

Determines whether an array contains a specified element:

let arr = [1, 2, 3, 4];
let isPresent = arr.includes(3);
console.log(isPresent); // true

15. sort()

Sorts the elements of an array in place and returns the sorted array:

let arr = [3, 1, 4, 2];
arr.sort();
console.log(arr); // [1, 2, 3, 4]

16. reverse()

Reverses the elements of an array in place:

let arr = [1, 2, 3, 4];
arr.reverse();
console.log(arr); // [4, 3, 2, 1]

17. join()

Joins all elements of an array into a single string:

let arr = ['Hello', 'World'];
let joined = arr.join(' ');
console.log(joined); // "Hello World"

18. some()

Tests whether at least one element in the array passes the test implemented by the provided function:

let arr = [1, 2, 3, 4];
let hasEven = arr.some(x => x % 2 === 0);
console.log(hasEven); // true

19. every()

Tests whether all elements in the array pass the test implemented by the provided function:

let arr = [1, 2, 3, 4];
let allEven = arr.every(x => x % 2 === 0);
console.log(allEven); // false
