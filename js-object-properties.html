OBJECT METHOD IN DETAILS

1. assing() :- The assign() method in JavaScript is used to copy the properties (or key-value pairs) from one or more source objects to a target object. It does not create a new object; instead, it modifies the target object. Syntax :- Object.assign(target, ...sources); target: Jisme properties copy hongi (ye change hoga). sources: Jisme se properties uthayi jayengi (1 ya zyada ho sakte hain). const target = { a: 1, b: 2 }; const source = { b: 4, c: 5 }; Object.assign(target, source); console.log(target); OUTPUT {a:1, b:4, c:5} 2. constructor :- The constructor property returns the function that created the Object prototype. Syntax :- Object.constructor Return Value :- function Object() { [native code] } 3. create() :- Create a new object from the prototype of another object. Syntax :- Object.create(object, properties) // Create an Object: const person = { firstName: "John", lastName: "Doe", age: 50, eyeColor: "blue" }; // Create new Object let man = Object.create(person); man.firstName = "Peter"; console.log(person.firstName, + " " + , man.firstName); 4. Object.defineProperties() :- This adds a new property to an object. Syntax :- Object.defineProperty(object, property, descriptor) object The object to edit. property The property to edit. descriptor A descriptor of the properties to be added or changed: value: value writable : true|false enumerable : true|false configurable : true|false get : function set : function const person = { firstName: "John", lastName: "Doe" }; Object.defineProperties(person, { language: {value: "en"}, year: {value: "Hello"} }); console.log(person.language + " " + person.year;)

1. defineProperty()

Ek naye ya existing property ko define ya modify karta hai:

let obj = {};
Object.defineProperty(obj, 'name', { value: 'Jack', writable: false });
console.log(obj.name); // "Jack"
obj.name = 'John'; // Change nahi hoga
console.log(obj.name); // "Jack"

2. entries()

Object ke key-value pairs ko array ke form me return karta hai:

let obj = { a: 1, b: 2, c: 3 };
console.log(Object.entries(obj)); 
// [['a', 1], ['b', 2], ['c', 3]]

3. freeze()

Object ko completely immutable bana deta hai:

let obj = { name: 'Jack' };
Object.freeze(obj);
obj.name = 'John'; // Change nahi hoga
console.log(obj.name); // "Jack"

4. fromEntries()

Array of key-value pairs ko object me convert karta hai:

let entries = [['a', 1], ['b', 2]];
let obj = Object.fromEntries(entries);
console.log(obj); // { a: 1, b: 2 }

5. getOwnPropertyDescriptor()

Property ka descriptor return karta hai:

let obj = { name: 'Jack' };
let desc = Object.getOwnPropertyDescriptor(obj, 'name');
console.log(desc); 
// { value: 'Jack', writable: true, enumerable: true, configurable: true }

6. getOwnPropertyDescriptors()

Saari properties ke descriptors ka object return karta hai:

let obj = { name: 'Jack', age: 25 };
console.log(Object.getOwnPropertyDescriptors(obj));

7. getOwnPropertyNames()

Object ki saari properties ke names return karta hai (including non-enumerable):

let obj = { a: 1, b: 2 };
console.log(Object.getOwnPropertyNames(obj)); // ['a', 'b']

8. groupBy()

Callback ke basis pe elements ko group karta hai:

let arr = ['one', 'two', 'three', 'four'];
let grouped = Object.groupBy(arr, word => word.length);
console.log(grouped);
// { 3: ['one', 'two'], 5: ['three'], 4: ['four'] }

9. isExtensible()

Check karta hai ki object me naye properties add ho sakti hain ya nahi:

let obj = { a: 1 };
console.log(Object.isExtensible(obj)); // true
Object.preventExtensions(obj);
console.log(Object.isExtensible(obj)); // false

10. isFrozen()

Check karta hai ki object freeze ho chuka hai ya nahi:

let obj = { a: 1 };
console.log(Object.isFrozen(obj)); // false
Object.freeze(obj);
console.log(Object.isFrozen(obj)); // true

11. isSealed()

Check karta hai ki object sealed hai ya nahi (matlab naye properties add/delete nahi ho sakti):

let obj = { a: 1 };
console.log(Object.isSealed(obj)); // false
Object.seal(obj);
console.log(Object.isSealed(obj)); // true

12. keys()

Object ki saari keys ka array return karta hai:

let obj = { a: 1, b: 2 };
console.log(Object.keys(obj)); // ['a', 'b']

13. preventExtensions()

Object ko extend hone se rokh deta hai (naye properties add nahi ho sakti):

let obj = { a: 1 };
Object.preventExtensions(obj);
obj.b = 2; // Add nahi hoga
console.log(obj); // { a: 1 }

14. prototype

Prototype chaining se naye methods/properties add karne ke liye use hota hai:

function Person(name) {
  this.name = name;
}
Person.prototype.greet = function () {
  console.log(`Hello, ${this.name}!`);
};
let p1 = new Person('Jack');
p1.greet(); // "Hello, Jack!"

15. seal()

Object me naye properties add ya delete nahi ho sakti, par existing properties modify ho sakti hain:

let obj = { a: 1 };
Object.seal(obj);
obj.a = 2; // Allowed
delete obj.a; // Not allowed
console.log(obj); // { a: 2 }

16. toString()

Object ko string me convert karta hai:

let obj = { a: 1 };
console.log(obj.toString()); // "[object Object]"

17. valueOf()

Object ka primitive value return karta hai:

let num = new Number(42);
console.log(num.valueOf()); // 42

18. values()

Object ke saare values ka array return karta hai:

let obj = { a: 1, b: 2 };
console.log(Object.values(obj)); // [1, 2]
